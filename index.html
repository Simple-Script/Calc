import math def triangle_area(base, height): return 0.5 * base * height def triangle_missing_height(area, base): return (2 * area) / base def triangle_missing_base(area, height): return (2 * area) / height def rectangle_area(length, width): return length * width def rectangle_missing_side(area, side, find_length=True): if find_length: return area / side else: return area / side def square_side_from_area(area): return math.sqrt(area) def rectangle_plot(length, width): area = length * width perimeter = 2 * (length + width) return area, perimeter # Example interactive program def main(): print("Geometry Solver") print("1. Triangle area") print("2. Missing height of triangle") print("3. Missing base of triangle") print("4. Rectangle area") print("5. Square side from area") print("6. Rectangle plot (area & fencing)") choice = int(input("Enter choice: ")) if choice == 1: b = float(input("Enter base: ")) h = float(input("Enter height: ")) print("Area =", triangle_area(b, h)) elif choice == 2: A = float(input("Enter area: ")) b = float(input("Enter base: ")) print("Height =", triangle_missing_height(A, b)) elif choice == 3: A = float(input("Enter area: ")) h = float(input("Enter height: ")) print("Base =", triangle_missing_base(A, h)) elif choice == 4: l = float(input("Enter length: ")) w = float(input("Enter width: ")) print("Area =", rectangle_area(l, w)) elif choice == 5: A = float(input("Enter area: ")) print("Side length =", square_side_from_area(A)) elif choice == 6: l = float(input("Enter length: ")) w = float(input("Enter width: ")) A, P = rectangle_plot(l, w) print("Area =", A, "Perimeter =", P) else: print("Invalid choice.") if __name__ == "__main__": main()
